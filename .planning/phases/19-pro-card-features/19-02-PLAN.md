    ```typescript
      id: number;
      username: string;
      slug: string;
      config: string; // JSON string
      created_at: number;

    }

    export interface SlugConfig {
      theme: string;       // e.g. "catppuccin", "dracula"
      layout: string;      // "classic" | "compact" | "hero"
      hide?: string[];     // stat keys to hide
      heroStat?: string;   // for hero layout
      colors?: {           // BYOT colors (PRO-only)
        bg: string;
        title: string;
        text: string;
        icon: string;
        border: string;
      };
    }
    ```

    **Constants:**
    ```typescript
    export const SLUG_MIN_LENGTH = 3;
    export const SLUG_MAX_LENGTH = 50;
    export const SLUG_MAX_PER_USER = 5;
    export const SLUG_REGEX = /^[a-z0-9][a-z0-9-]*[a-z0-9]$/;
    export const SLUG_RESERVED = new Set([
      "admin", "api", "settings", "config", "dashboard",
      "billing", "sync", "auth", "webhook", "community",
      "configure", "login", "logout", "help", "support",
      "pro", "free", "upgrade", "pricing",
    ]);
    ```

    **Validation function:**
    ```typescript
    export function validateSlug(slug: string): string | null {
      if (slug.length < SLUG_MIN_LENGTH) return "Slug must be at least 3 characters";
      if (slug.length > SLUG_MAX_LENGTH) return "Slug must be 50 characters or fewer";
      if (!SLUG_REGEX.test(slug)) {
        return "Slug must be lowercase alphanumeric with hyphens only (no leading/trailing hyphens)";
      }
      if (SLUG_RESERVED.has(slug)) return `"${slug}" is a reserved word`;
      return null; // valid
    }
    ```
    Returns null if valid, error message string if invalid.

    **Query helpers (all take `db: D1Database` as first param):**

    `getUserSlugs(db, username)` — SELECT * FROM card_slugs WHERE username = ? ORDER BY created_at ASC. Returns CardSlug[].

    `getSlug(db, username, slug)` — SELECT * FROM card_slugs WHERE username = ? AND slug = ?. Returns CardSlug | null.

    `countUserSlugs(db, username)` — SELECT COUNT(*) as count FROM card_slugs WHERE username = ?. Returns number.

    `createSlug(db, username, slug, config: SlugConfig)` — INSERT INTO card_slugs (username, slug, config) VALUES (?, ?, ?). Binds JSON.stringify(config). Returns void.

    `deleteSlug(db, username, slug)` — DELETE FROM card_slugs WHERE username = ? AND slug = ?. Returns void.

    Export everything: types, constants, validateSlug, and all 5 query functions.
