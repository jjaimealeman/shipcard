    **Card route — add slug serving:**

    In `card.ts`, add a NEW route handler BEFORE the existing `/:username` handler:

    ```typescript
    cardRoutes.get("/:username/:slug", async (c) => {
      const username = c.req.param("username");
      const slug = c.req.param("slug");

      // 1. Load slug config from D1
      const slugRow = await getSlug(c.env.DB, username, slug);
      if (!slugRow) {
        return svgResponse(c, renderPlaceholderCard(username));
      }

      // 2. Check KV cache for this slug
      const cacheKey = `card:${username}:slug:${slug}`;
      const cached = await c.env.CARDS_KV.get(cacheKey);
      if (cached) return svgResponse(c, cached);

      // 3. Load user data
      const userData = await getUserData(c.env.USER_DATA_KV, username);
      if (!userData) return svgResponse(c, renderPlaceholderCard(username));

      // 4. Parse config and render
      const config = JSON.parse(slugRow.config) as SlugConfig;
      const isPro = await isUserPro(c.env.DB, username);

      // Resolve colors: if config has BYOT colors, use them; otherwise resolve curated theme
      let colors: ThemeColors | undefined;
      if (config.colors) {
        colors = {
          bg: config.colors.bg,
          border: config.colors.border,
          title: config.colors.title,
          text: config.colors.text,
          value: config.colors.title,   // value = title
          icon: config.colors.icon,
          footer: config.colors.text,   // footer = text
        };
      } else {
        const curated = resolveCuratedTheme(config.theme);
        if (curated) colors = curated;
      }

      const svg = renderCard(userData, {
        layout: (config.layout || "classic") as LayoutName,
        colors,
        hide: config.hide,
        heroStat: config.heroStat,
        isPro,
      });

      // 5. Cache the rendered SVG
      await c.env.CARDS_KV.put(cacheKey, svg);
      return svgResponse(c, svg);
    });
    ```

    Add the necessary imports at the top of card.ts:

    - `import type { SlugConfig } from "../db/slugs.js";`

    **CRITICAL:** The `/:username/:slug` route MUST be defined BEFORE the existing `/:username` route in card.ts. Hono matches by registration order — two-segment paths must be registered first.

    ```typescript
      for (const slugRow of slugs) {
        const config = JSON.parse(slugRow.config) as SlugConfig;

        if (config.colors) {
          colors = {

            title: config.colors.title,
            text: config.colors.text,

            icon: config.colors.icon,
            footer: config.colors.text,
          };
        } else {
          const curated = resolveCuratedTheme(config.theme);
          if (curated) colors = curated;
        }

        const slugSvg = renderCard(body, {
          layout: (config.layout || "classic") as LayoutName,
          colors,
          hide: config.hide,
          heroStat: config.heroStat,
          isPro: true,
        });
        const cacheKey = `card:${username}:slug:${slugRow.slug}`;
        await env.CARDS_KV.put(cacheKey, slugSvg);
      }
    }
    ```
