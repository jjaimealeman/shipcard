**Alpine.js global store (`Alpine.store('dashboard')`):**
```javascript
Alpine.store('dashboard', {
  username: '__USERNAME__',
  range: '30d',        // default
  stats: null,         // SafeStats from API
  timeseries: null,    // SafeTimeSeries from API
  loading: true,
  error: null,

  // Computed: filtered days based on range
  get filteredDays() {
    if (!this.timeseries || !this.timeseries.days) return [];
    if (this.range === 'all') return this.timeseries.days;
    const now = new Date();
    const cutoff = new Date(now);
    cutoff.setDate(cutoff.getDate() - (this.range === '7d' ? 7 : 30));
    const cutoffStr = cutoff.toISOString().slice(0, 10);
    return this.timeseries.days.filter(d => d.date >= cutoffStr);
  },

  // Computed: has project data
  get hasProjects() {
    if (!this.timeseries || !this.timeseries.days) return false;
    return this.timeseries.days.some(d => d.projects && d.projects.length > 0);
  },

  async init() {
    try {
      const [statsRes, tsRes] = await Promise.all([
        fetch('/u/' + this.username + '/api/stats'),
        fetch('/u/' + this.username + '/api/timeseries')
      ]);
      if (statsRes.ok) this.stats = (await statsRes.json()).data;
      if (tsRes.ok) this.timeseries = (await tsRes.json()).data;
      if (!statsRes.ok && !tsRes.ok) this.error = 'User not found';
    } catch (e) {
      this.error = 'Failed to load data';
    }
    this.loading = false;
  }
});
```
