**Data aggregation helpers (define as functions in script):**
```javascript
function aggregateByWeekday(days) {
  const result = Array(7).fill(0); // Mon=0 ... Sun=6
  days.forEach(d => {
    const dow = new Date(d.date + 'T12:00:00').getDay(); // 0=Sun
    const idx = dow === 0 ? 6 : dow - 1; // shift to Mon=0
    result[idx] += d.sessions;
  });
  return result;
}

function aggregateField(days, field) {
  const result = {};
  days.forEach(d => {
    const map = d[field] || {};
    Object.entries(map).forEach(([k, v]) => {
      result[k] = (result[k] || 0) + v;
    });
  });
  return result;
}

function topN(obj, n) {
  const sorted = Object.entries(obj).sort((a,b) => b[1] - a[1]);
  if (sorted.length <= n) return { labels: sorted.map(e=>e[0]), values: sorted.map(e=>e[1]) };
  const top = sorted.slice(0, n);
  const other = sorted.slice(n).reduce((sum, e) => sum + e[1], 0);
  return {
    labels: [...top.map(e=>e[0]), 'Other'],
    values: [...top.map(e=>e[1]), other]
  };
}
```

**Chart creation pattern — create once, update on range change:**

Store chart instances in a `charts` object. After Alpine initializes and data loads:

```javascript
document.addEventListener('alpine:init', () => {
  // Wait for data to load, then create charts
  Alpine.effect(() => {
    const store = Alpine.store('dashboard');
    if (store.loading || store.error) return;

    const days = store.filteredDays;
    if (!days.length) return;

    // Create or update each chart
    updateActivityChart(days);
    updateDailyChart(days);
    // ... etc
  });
});
```

**Datalabels config for donuts:**
```javascript
plugins: {
  datalabels: {
    color: COLORS.fg,
    font: { size: 11, weight: '600' },
    formatter: (value, ctx) => {
      const total = ctx.dataset.data.reduce((a,b) => a+b, 0);
      const pct = Math.round(value / total * 100);
      if (pct < 5) return ''; // hide small segments
      const label = ctx.chart.data.labels[ctx.dataIndex];
      return label + ': ' + value.toLocaleString() + ' (' + pct + '%)';
    }
  }
}
```
