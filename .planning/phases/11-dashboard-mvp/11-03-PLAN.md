    groupY: 'sum'

  date: {
    start: startDate,

    }
  },
  domain: {
    type: 'month',
    gutter: 4,
    label: { text: 'MMM', textAlign: 'start', position: 'top' }

  subDomain: {
    type: 'day',
    radius: 2,
    width: 12,
    height: 12,
    gutter: 2
  },
  itemSelector: '#heatmap-container'
});
