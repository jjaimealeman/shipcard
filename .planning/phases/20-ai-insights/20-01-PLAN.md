```typescript
export interface PeakHoursInsight {
  // 24 buckets (index 0=midnight, 23=11pm) with total sessions per hour
  hourlyTotals: number[];
  // Top 3 hours sorted by activity
  topHours: Array<{ hour: number; label: string; totalSessions: number }>;
}

export interface PeakDaysInsight {
  // Top 3 days-of-week by average sessions
  topDays: Array<{ dayOfWeek: number; label: string; avgSessions: number }>;
}

export interface CostTrendInsight {
  // Weekly cost totals. Free: 2 entries, PRO: 4 entries
  weeklyTotals: Array<{ weekStart: string; costCents: number }>;
  trend: "up" | "down" | "flat";
  deltaPercent: number;
}

export interface StreakInsight {
  currentStreak: number;
  longestStreak: number;
  activeDaysThisWeek: number;
}

export interface InsightResult {
  username: string;
  computedAt: string;        // ISO timestamp
  isPro: boolean;
  windowDays: number;        // 14 for free, 28 for PRO
  peakHours?: PeakHoursInsight;  // Only present when hourlyActivity data is available
  peakDays: PeakDaysInsight;
  costTrend: CostTrendInsight;
  streak: StreakInsight;
  narrative?: string;           // PRO only — Workers AI generated
  narrativeError?: boolean;     // true if AI call failed
}
```

Create `shipcard-worker/src/insights/compute.ts` with these pure functions:

1. `getWeekStart(dateStr: string): string` — Returns Monday ISO date for the week containing dateStr. Use `getUTCDay()` to find offset from Monday.

2. `computePeakHours(days: SafeDailyStats[]): PeakHoursInsight | undefined` — Takes days that have `hourlyActivity` field. If no day has hourlyActivity, return undefined. Sum all hourlyActivity arrays into 24 buckets. Return top 3 hours with labels like "9 AM", "2 PM".

3. `computePeakDays(days: SafeDailyStats[]): PeakDaysInsight` — Bucket days by day-of-week using `new Date(day.date + "T12:00:00Z").getUTCDay()` (noon UTC avoids DST edge). For each DOW, compute average sessions. Return top 3 by avgSessions. Day labels: ["Sun", "Mon", "Tue", "Wed", "Thu", "Fri", "Sat"].

4. `computeCostTrend(days: SafeDailyStats[], windowDays: number): CostTrendInsight` — Group days into ISO weeks (Mon-Sun). Trim to last `windowDays / 7` weeks (2 for free, 4 for PRO). Compute costCents sum per week. Trend = compare last two weeks: up if delta > 5%, down if < -5%, flat otherwise. deltaPercent = ((last - prev) / prev) * 100 (handle prev=0 as "up" if last > 0, "flat" if both 0).

5. `computeStreak(days: SafeDailyStats[]): StreakInsight` — Use the algorithm from RESEARCH.md. Active day = costCents > 0. Current streak: walk backward from today. Longest streak: linear scan. activeDaysThisWeek: count days in current Mon-Sun week with costCents > 0.

6. `computeAllInsights(days: SafeDailyStats[], username: string, isPro: boolean): InsightResult` — Orchestrator. Determines windowDays (14 free, 28 PRO). Filters days to the window (last N calendar days from today). Calls all compute functions. Returns assembled InsightResult with computedAt = now ISO.

Import `SafeDailyStats` from `../types.js`. Do NOT use `any` — all types are explicit.

```typescript
import type { InsightResult } from "./types.js";

/**
 * Build a prompt for Workers AI to generate a 2-3 sentence narrative
 * summarizing the user's weekly coding insights.
 */
export function buildNarrativePrompt(insights: InsightResult): string {
  // Build a data summary string from the insight stats.
  // Include: top peak days, cost trend direction + delta, current streak, active days.
  // If peakHours is available, include top coding hour.
  // Keep prompt under ~400 tokens for efficiency.
  // Format: "Here are the weekly coding stats for {username}: ..."
  // Ask for a 2-3 sentence narrative summary.
}

/**
 * Call Workers AI to generate a narrative for the insight result.
 * Uses @cf/meta/llama-3.2-1b-instruct with max_tokens:150.
 * Returns the narrative string, or undefined if the call fails.
 */
export async function callWorkersAI(
  ai: Ai,
  insights: InsightResult
): Promise<string | undefined> {
  try {
    const prompt = buildNarrativePrompt(insights);
    const result = await ai.run(
      "@cf/meta/llama-3.2-1b-instruct" as BaseAiTextGenerationModels,
      {
        messages: [
          {
            role: "system",
            content: "You are a concise coding activity analyst. Write exactly 2-3 sentences summarizing the developer's weekly activity. Be encouraging but factual. Do not use markdown formatting."
          },
          { role: "user", content: prompt }
        ],
        max_tokens: 150,
      }
    );
    // Workers AI returns { response: string } for text generation
    const response = (result as { response?: string }).response;
    return response?.trim() || undefined;
  } catch {
    return undefined;
  }
}
```

The `Ai` type and `BaseAiTextGenerationModels` are globally available from `@cloudflare/workers-types`. Do NOT import them — they are ambient types.

For `buildNarrativePrompt`, construct a clean text string like:

```
Weekly coding stats for {username} ({windowDays}-day window):
- Most active days: {topDay1}, {topDay2}, {topDay3}
- Cost trend: {trend} ({deltaPercent}% vs previous week)
- Current coding streak: {currentStreak} days
- Active days this week: {activeDaysThisWeek}/7
- Longest streak: {longestStreak} days
{If peakHours: "- Peak coding hour: {topHour}"}

Write a 2-3 sentence narrative summary of this developer's activity.
```
