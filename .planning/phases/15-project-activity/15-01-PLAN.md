```javascript
get _projectTotal() {
  const all = this.timeseries ? this.timeseries.days : [];
  const names = new Set();
  all.forEach(d => {
    if (d.byProject) Object.keys(d.byProject).forEach(n => names.add(n));
    if (d.projects) d.projects.forEach(n => names.add(n));
  });
  return names.size;
},
```
